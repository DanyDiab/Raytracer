cmake_minimum_required(VERSION 3.15)
project(Raytracer VERSION 1.0 LANGUAGES CXX CUDA)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CUDA_STANDARD 17)
set(CMAKE_CUDA_STANDARD_REQUIRED ON)

set(SCRIPTSDIR "./scripts")

find_package(glfw3 3.3.10 REQUIRED)

add_executable(Raytracer 
"${SCRIPTSDIR}/Window.cpp"
"${SCRIPTSDIR}/Main.cpp"
"${SCRIPTSDIR}/Camera.cu")

target_include_directories(Raytracer PRIVATE "${SCRIPTSDIR}/headers")

target_link_libraries(Raytracer PRIVATE glfw)
target_link_libraries(Raytracer PRIVATE cuda cudart)

target_compile_options(Raytracer PRIVATE -g)
